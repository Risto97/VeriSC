set(__LIB_NAME systemc-ams)

set(SYSTEMC-AMS_INSTALL_DIR ${GROUP_INSTALL_DIR}/${__LIB_NAME}-${SYSTEMC-AMS_VERSION})
set(SYSTEMC-AMS_SRC_DIR ${PROJECT_BINARY_DIR}/src/${__LIB_NAME})

ExternalProject_Add(systemc-ams
    GIT_REPOSITORY https://github.com/Risto97/systemc-ams.git
    GIT_TAG ${SYSTEMC-AMS_VERSION}
    PREFIX ${PROJECT_BINARY_DIR}
    INSTALL_DIR ${SYSTEMC-AMS_INSTALL_DIR}
    CMAKE_ARGS
    -DCMAKE_INSTALL_PREFIX=${SYSTEMC-AMS_INSTALL_DIR}
        -DCMAKE_INSTALL_LIBDIR=lib
        -DSYSTEMC_HOME=${SYSTEMC_INSTALL_DIR}
        -DCMAKE_CXX_STANDARD=${CMAKE_CXX_STANDARD}
        -DCMAKE_CXX_COMPILER=${CMAKE_CXX_COMPILER}
        -DCMAKE_C_COMPILER=${CMAKE_C_COMPILER}
    DEPENDS systemc
    )
